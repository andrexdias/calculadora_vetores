111def misturar_vetores(vetor1, vetor2):
112 """
113 Mistura metade do primeiro vetor com metade do segundo vetor.
114 """
115 meio1 = len(vetor1) // 2

119def numeros_compostos(vetor):
120 """
121 Identifica todos os números compostos (não primos) no vetor inicial.
122 """
123 compostos = [valor for valor in vetor if any(valor % i == 0 for i in range(2, valor))]
124 return compostos

126def produto_matriz(vetor1, vetor2):
127 """
128 Calcula e retorna a matriz resultante do produto do vetor1 com o vetor2.
129 """
130 matriz = [[a * b for b in vetor2] for a in vetor1]

133def matriz_transposta(matriz):
134 """
135 Calcula e retorna a matriz transposta da matriz fornecida.
136 """
137 transposta = [[matriz[j][i] for j in range(len(matriz))] for i in range(len(matriz[0]))]
138 return transposta
